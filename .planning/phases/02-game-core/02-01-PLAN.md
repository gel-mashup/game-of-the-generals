# Battle Resolution — THE critical algorithm
# Priority order (MUST be followed exactly):

function resolveBattle(attacker: Piece, defender: Piece): BattleOutcome:
  // Rule 1: Flag capture — any piece captures flag
  if defender.type === 'flag':
    return { winner: attacker.owner, captured: defender }

  // Rule 2: Spy vs Private (MUST come before rank comparison)
  if attacker.type === 'spy' and defender.type === 'private':
    return { winner: defender.owner, captured: attacker }  // Private wins
  if attacker.type === 'private' and defender.type === 'spy':
    return { winner: attacker.owner, captured: defender }  // Private wins

  // Rule 3: Equal rank — both eliminated
  if attacker.rank === defender.rank:
    return { winner: 'tie', captured: [attacker, defender] }

  // Rule 4: Higher rank wins
  if attacker.rank > defender.rank:
    return { winner: attacker.owner, captured: defender }
  else:
    return { winner: defender.owner, captured: attacker }

# Rank values (from types):
# 5-star: 11, 4-star: 10, 3-star: 9, 2-star: 8, 1-star: 7,
# colonel: 6, lieutenant-colonel: 5, major: 4, captain: 3,
# 1st-lieutenant: 2, 2nd-lieutenant: 1, sergeant: 0,
# private: -1, spy: -2, flag: -3

       - Must place exactly 21 pieces (respecting PIECE_CONFIG counts)
